## How to build a Whole Cell pyenv virtualenv:
## Prepare as needed. Install these packages using your local package
## manager such as brew on macOS:
#   brew install pyenv
#   brew install pyenv-virtualenv
#   brew install pyenv-virtualenvwrapper
#   brew install glpk
#   brew install swig           # needed to install pip CVXcanon
#   brew install suite-sparse   # needed for cvxopt to use glpk
#
## Download and install other packages according to their instructions
#	CPLEX from IBM (free for students)
#
## Create a pyenv with Python 2.7.14.
#   PYTHON_CONFIGURE_OPTS="--enable-shared" pyenv install 2.7.14  # if needed
#   pyenv local 2.7.14
#
## Create and select a new python virtual environment:
#   pyenv virtualenv wcEcoli2
#   pyenv local wcEcoli2
#
## Upgrade the installers but don't use pip==10.0.0 yet because it
## breaks cobra test.
#   pip install --upgrade pip==9.0.3 setuptools virtualenv wheel
#
## Install requires numpy before running requirements.txt
#   pip install numpy==1.14.3
#
## For Sherlock:
## See /share/PI/mcovert/installation_notes/scipy_with_openblas.txt on how
## to build scipy from source using a site.cfg file that contains:
##
##      [atlas]
##      atlas_libs = openblas
##      libraries = openblas
##
##      [openblas]
##      libraries = openblas
##      library_dirs = /share/sw/free/openblas/0.2.15/lib
##      include_dirs = /share/sw/free/openblas/0.2.15/include
#
## Then install all the other packages listed here:
#   CVXOPT_BUILD_GLPK=1 pip install -r requirements.txt --no-binary cvxopt
#   pyenv rehash
#
# (If an older version of pip or python gives an error that numpy isn't
# installed, install numpy then retry installing requirements.txt.)
